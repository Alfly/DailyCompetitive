## [HJ69 矩阵乘法](https://www.nowcoder.com/practice/ebe941260f8c4210aa8c17e99cbc663b?tpId=37&tqId=21292&ru=/exam/oj)

> Tag: 数学 模拟

### 方法一：模拟
* 时间复杂度: ${O(m * n * p)}$
* 空间复杂度: ${O(1)}$
```cpp
#include <iostream>
#include <vector>
using namespace std;

int main() {
    int m, p, n;
    cin >> m >> p >> n;
    vector<vector<int> > A(m, vector<int>(p));
    vector<vector<int> > B(p, vector<int>(n));
    vector<vector<int> > C(m, vector<int>(n));

    for (int i = 0; i < m; i++) {
        for (int j = 0; j < p; j++) {
            cin >> A[i][j];
        }
    }

    for (int i = 0; i < p; i++) {
        for (int j = 0; j < n; j++) {
            cin >> B[i][j];
        }
    }
    
    for (int i = 0; i < m; i++) {
        for (int j = 0; j < n; j++) {
            for (int k = 0; k < p; k++) {
                C[i][j] += A[i][k] * B[k][j];
            }
            cout << C[i][j] << ' ';
        }
        cout << endl;
    }
    return 0;
}
```
