## [54. 螺旋矩阵](https://leetcode.cn/problems/spiral-matrix/description/)

> Tag: 模拟

### 方法一：模拟-设定边界
* 时间复杂度: ${O(n^2)}$
* 空间复杂度: ${O(1)}$

**写法一：每次边界更新，立即判断是否继续**

```cpp
class Solution {
public:
    vector<int> spiralOrder(vector<vector<int>>& matrix) {
        vector<int> ans;
        int m = matrix.size(), n = matrix[0].size();
        int up = 0, down = m - 1;
        int left = 0, right = n - 1;
        int i = 0, j = 0;
        while (1) {
            for (j = left; j <= right; j++) ans.push_back(matrix[up][j]);
            if (++up > down) break;
            
            for (i = up; i <= down; i++) ans.push_back(matrix[i][right]);
            if (left > --right) break;

            for (j = right; j >= left; j--) ans.push_back(matrix[down][j]);
            if (up > --down) break;

            for (i = down; i >= up; i--) ans.push_back(matrix[i][left]);
            if (++left > right) break;
        }
        return ans;
    }
};
```

while 控制 i j 的写法：较复杂

```cpp
class Solution {
public:
    vector<int> spiralOrder(vector<vector<int>>& matrix) {
        vector<int> ans;
        int m = matrix.size(), n = matrix[0].size();
        int up = 0, down = m - 1;
        int left = 0, right = n - 1;
        int i = 0, j = 0;
        while (1) {
            while (j <= right) ans.push_back(matrix[i][j++]);
            j--;
            if (++up > down) break;

            i++;
            while (i <= down) ans.push_back(matrix[i++][j]);
            if (--right < left) break;
            i--;

            j--;
            while (j >= left) ans.push_back(matrix[i][j--]);
            if (--down < up) break;
            j++;

            i--;
            while (i >= up) ans.push_back(matrix[i--][j]);
            if (++left > right) break;
            i++;
            j++;
        }
        return ans;
    }
};
```

**写法二：两次移动一次判断**

若不在 [右下] [左上]添加一个边界判断，当矩阵为扁形，会多一次移动。所以至少两次判断 (1)[右下] (2)[左上]

```cpp
class Solution {
public:
    vector<int> spiralOrder(vector<vector<int>>& matrix) {
        int up = 0, left = 0, down = matrix.size() - 1, right = matrix[0].size() - 1;
        vector<int> ans;
        int i = 0;
        while (left <= right && up <= down) {
            for (i = left; i <= right; i++) ans.push_back(matrix[up][i]);
            ++up;
            
            for (i = up; i <= down; i++) ans.push_back(matrix[i][right]);
            --right;
            
            // 注意判断
            if (left > right || up > down) break;
            
            for (i = right; i >= left; i--) ans.push_back(matrix[down][i]);
            --down;
            
            for (i = down; i >= up; i--) ans.push_back(matrix[i][left]);
            ++left;
        }
        return ans;
    }
};
```
