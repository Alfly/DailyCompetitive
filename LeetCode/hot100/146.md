## [146. LRU 缓存](https://leetcode.cn/problems/lru-cache/description/)

> Tag: LRU 缓存算法 哈希 双向链表

哈希表用于快速查找缓存项，双向链表用于维护缓存项的访问顺序。

每当缓存项被访问时，可以通过哈希表快速定位到对应的链表节点，并将该节点移到链表头部。

### 方法一：哈希表+双向链表

朴素版本：双向链表手动实现，主要是 `remove` 和 `add2head` 两个函数

KV 节点使用双向链表链接，为了方便操作，定义两个哨兵节点 `head` 和 `tail`

缓存 cache 是哈希表 `{key: Node*}`，节点中存放 key 是为了方便删除。

```cpp
    struct Node {
        int key, value;
        Node *next, *pre;
        Node() {}
        Node(int x, int y) { key = x; value = y; }
    };
    unordered_map<int, Node*> cache;
    Node *head, *tail;
    int capacity, size = 0;
```

```cpp
class LRUCache {
private:
    struct Node {
        int key, value;
        Node *next, *pre;
        Node() {}
        Node(int x, int y) { key = x; value = y; }
    };
    unordered_map<int, Node*> cache;
    Node *head, *tail;
    int capacity, size = 0;

public:
    LRUCache(int capacity) {
        this->capacity = capacity;
        head = new Node(), tail = new Node();
        head->next = tail, tail->pre = head;
    }
    
    int get(int key) {
        if (!cache.count(key)) {
            return -1;
        }

        // 将访问的节点移动到链表头部
        Node *cur = cache[key];
        move2head(cur);

        return cur->value;
    }
    
    void put(int key, int value) {
        if (!cache.count(key)) {
            // 向缓存添加新的 key
            if (size >= capacity) {
                // 如果容量已满，移除最久未使用的节点
                Node *lastKeyVal = removetail();
                cache.erase(lastKeyVal->key);
                size--;
            }
            
            // 将新节点添加到链表头部
            Node *cur = new Node(key, value);
            add2head(cur);
            cache[key] = cur;
            size++;
        } else {
            // 如果 key 已存在，更新节点的值，并将节点移动到链表头部
            Node *cur = cache[key];
            cur->value = value;
            move2head(cur);
        }
    }

    // Biderection List 双向链表操作
    void remove(Node* t) {
        t->pre->next = t->next;
        t->next->pre = t->pre;
    }
    void add2head(Node* t) {
        t->pre = head;
        t->next = head->next;
        head->next->pre = t;
        head->next = t;
    }
    void move2head(Node* t) {
        remove(t);
        add2head(t);
    }
    Node* removetail() {
        Node *t = tail->pre;
        remove(t);
        return t;
    }
};

/**
 * Your LRUCache object will be instantiated and called as such:
 * LRUCache* obj = new LRUCache(capacity);
 * int param_1 = obj->get(key);
 * obj->put(key,value);
 */
```

双向链表使用 STL 的 `list` 实现

```c++
class LRUCache {
private:
    int capacity;
    list<pair<int, int>> lrulist;
    unordered_map<int, list<pair<int, int>>::iterator> cache;

public:
    LRUCache(int capacity) {
        this->capacity = capacity;
    }
    
    int get(int key) {
        if (cache.find(key) == cache.end()) {
            return -1; // 如果 key 不存在，返回 -1
        }
        // 将访问的节点移动到链表头部
        pair<int, int> curKeyVal = *cache[key];
        lrulist.erase(cache[key]);
        lrulist.push_front(curKeyVal);
        cache[key] = lrulist.begin();

        return curKeyVal.second;
    }
    
    void put(int key, int value) {
        if (cache.find(key) == cache.end()) {
            // 向缓存添加新的 key
            if (lrulist.size() == capacity) {
                // 如果容量已满，移除最久未使用的节点
                pair<int, int> lastKeyVal = lrulist.back();
                int lastKey = lastKeyVal.first;
                cache.erase(lastKey);
                lrulist.pop_back();
            }

            // 将新节点添加到链表头部
            lrulist.push_front(make_pair(key, value));
            cache[key] = lrulist.begin();
        } else {
            // 如果 key 已存在，更新节点的值，并将节点移动到链表头部
            lrulist.erase(cache[key]);
            lrulist.push_front(make_pair(key, value));
            cache[key] = lrulist.begin();
        }
    }
};

/**
 * Your LRUCache object will be instantiated and called as such:
 * LRUCache* obj = new LRUCache(capacity);
 * int param_1 = obj->get(key);
 * obj->put(key,value);
 */
```
