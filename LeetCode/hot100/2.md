## [2. 两数相加](https://leetcode.cn/problems/add-two-numbers/description)

> Tag: 链表

### 方法一：加法模拟
* 时间复杂度: ${O(max(m, n))}$
* 空间复杂度: ${O(1)}$

直接新建一个链表，链表头是哨兵节点

```cpp
class Solution {
public:
    ListNode* addTwoNumbers(ListNode* l1, ListNode* l2) {
        ListNode* h = new ListNode(0), *p = h;
        int t = 0;
        while (l1 && l2) {
            t += l1->val + l2->val;
            p->next = new ListNode(t % 10);;
            p = p->next;
            t /= 10;
            l1 = l1->next, l2 = l2->next;
        }
        ListNode* q = l1 ? l1 : l2;
        while (q) {
            t += q->val;
            p->next = new ListNode(t % 10);
            p = p->next;
            t /= 10;
            q = q->next;
        }
        if (t) {
            p->next = new ListNode(t);
        }
        return h->next;
    }
};
```

class Solution {

};

不使用哨兵头节点，在初始插入时需要判断头节点是否为空

```cpp
class Solution {
public:
    ListNode* addTwoNumbers(ListNode* l1, ListNode* l2) {
        ListNode *h = nullptr, *p = nullptr;
        int t = 0;
        while (l1 || l2) {
            if (l1) t += l1->val;
            if (l2) t += l2->val;
            if (!h) {
                h = p = new ListNode(t % 10);
            } else {
                p->next = new ListNode(t % 10);
                p = p->next;
            }
            t /= 10;
            if (l1) l1 = l1->next;
            if (l2) l2 = l2->next;
        }
        if (t) {
            p->next = new ListNode(t);
        }
        return h;
    }
};
```

**只需要开辟一次新空间**的写法，实际并没有快多少

```cpp
class Solution {
public:
    ListNode* addTwoNumbers(ListNode* l1, ListNode* l2) {
        ListNode *p1 = l1, *p2 = l2;
        int cnt1 = 0, cnt2 = 0;
        while (p1) {
            cnt1++;
            p1 = p1->next;
        }
        while (p2) {
            cnt2++;
            p2 = p2->next;
        }
        if (cnt1 > cnt2) return addTwoNumbers(l2, l1);

        int t = 0;
        p1 = l1, p2 = l2;
        ListNode *pre;
        while (p1 && p2) {
            t += p1->val + p2->val;
            p2->val = t % 10;
            t /= 10;
            pre = p2;
            p1 = p1->next, p2 = p2->next;
        }
        while (p2) {
            t += p2->val;
            p2->val = t % 10;
            t /= 10;
            pre = p2;
            p2 = p2->next;
        }
        if (t) {
            pre->next = new ListNode(t);
        }
        return l2;
    }
};
```
