## [79. 单词搜索](https://leetcode-cn.com/problems/word-search/)

> Tag: 回溯

### 方法一：回溯

注意及时剪枝，可以在原数组中添加#标记来替代 `st[][]`

* 时间复杂度: ${O()}$
* 空间复杂度: ${O()}$
```cpp
class Solution {
public:
    int xx[4] = {0, 0, 1, -1};
    int yy[4] = {1, -1, 0, 0};
    int m, n;
    vector<vector<bool>> st;
    bool exist(vector<vector<char>>& board, string word) {
        m = board.size(), n = board[0].size();
        st = vector<vector<bool>>(m, vector<bool>(n));
        // 遍历起点
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                if (dfs(board, word, i, j, 0)) return true;
            }
        }
        return false;
    }
    bool dfs(vector<vector<char>>& board, string word, int x, int y, int u) {
        if (u == word.size()) return true;
        if (x < 0 || x >= m || y < 0 || y >= n || st[x][y]) return false;
        // 剪枝
        if (board[x][y] != word[u]) return false;
        bool res = false;
        st[x][y] = true;
        for (int d = 0; d < 4; d++) {
            int a = x + xx[d], b = y + yy[d];
            if (dfs(board, word, a, b, u + 1)) return res = true;
        }
        st[x][y] = false;
        return res;
    }
};
```

内部函数

```cpp
class Solution {
public:
    int xx[4] = {0, 0, 1, -1};
    int yy[4] = {1, -1, 0, 0};
    bool exist(vector<vector<char>>& board, string word) {
        int m = board.size(), n = board[0].size();
        vector<vector<bool>> st(m, vector<bool>(n));

        function<bool(int, int, int)> dfs = [&](int i, int j, int u) -> bool {
            if (u == word.size()) return true;
            if (i < 0 || i >= m || j < 0 || j >= n) return false;
            if (st[i][j] || board[i][j] != word[u]) return false;
            st[i][j] = true;
            bool ans = false;
            for (int d = 0; d < 4; d++) {
                int x = i + xx[d], y = j + yy[d];
                // if (dfs(x, y, u + 1)) return ans = true;
                ans |= dfs(x, y, u + 1);
            }
            st[i][j] = false;
            return ans;
        };

        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                if (dfs(i, j, 0)) return true;
            }
        }
        return false;
    }
};
```

不回溯的**错误写法**(每个出发点只走一条路)

```cpp
class Solution {
public:
    int m, n;
    vector<vector<bool>>st;
    bool exist(vector<vector<char>>& board, string word) {
        m = board.size(), n = board[0].size();
        st.resize(m, vector<bool>(n));
        for (int i = 0; i < m; i ++) {
            for (int j = 0; j < n; j ++) {
                for (int k = 0; k < m; k ++) fill(st[k].begin(), st[k].end(), false);
                if (dfs(i, j, board, 0, word)) return true;
            }
        }
        return false;
    }
    bool dfs(int x, int y, vector<vector<char>>& board, int u, string& word) {
        if (u == word.size()) return true;
        if (x < 0 || x >= m || y < 0 || y >= n || st[x][y]) return false;
        if (board[x][y] != word[u]) return false;
        st[x][y] = true;
        return (dfs(x + 1, y, board, u + 1, word)) ||
               (dfs(x - 1, y, board, u + 1, word)) ||
               (dfs(x, y + 1, board, u + 1, word)) ||
               (dfs(x, y - 1, board, u + 1, word));
    }
};
```
