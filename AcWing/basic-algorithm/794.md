## [794. 高精度除法](https://www.acwing.com/problem/content/796/)

> Tag: 高精度

### 方法一：数组模拟
* 时间复杂度: ${O(n)}$
* 空间复杂度: ${O(n)}$
```c++
#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

// A / b = C ... r, A >= 0, b > 0
vector<int> div(vector<int> &A, int b, int& r) {
    vector<int> C;
    
    // 高位开始除法
    r = 0;
    for (int i = A.size() - 1; i >= 0; i--) {
        r = r * 10 + A[i];
        C.push_back(r / b);
        r %= b;
    }
    
    // 逆序，使数组从低位开始
    reverse(C.begin(), C.end());

    // 去除前导 0（b = 0 的情况）
    while(C.size() > 1 && C.back() == 0) C.pop_back();
     
    return C;
}

int main() {
    string a;
    int b;
    cin >> a >> b;
    
    vector<int> A;
    for(int i = a.size() - 1; i >= 0; i --) A.push_back(a[i] - '0');  
    
    int r = 0;
    auto C = div(A, b, r);
    for(int i = C.size() - 1; i >= 0; i --) cout << C[i];
    cout << '\n' << r;
    
    return 0;
}
```
